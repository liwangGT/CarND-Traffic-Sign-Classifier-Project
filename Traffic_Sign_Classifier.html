### Augment data set (using rotation, shift, change contrast)
from PIL import Image
import random
# rotate original image by degree
def Rotation(img, angle):
    img2 = Image.fromarray(img)
    img3 = img2.rotate(angle)
    imgout = np.array(img3)
    return imgout

# shift original image by dx, by
def Shift(img, dx, dy):
    imgout = np.zeros(img.shape)
    if dx>0 and dy>0:
        imgout[dx:, dy:, :] = img[:-dx, :-dy, :]
    if dx>0 and dy<0:
        imgout[dx:, :dy, :] = img[:-dx, -dy:, :]
    if dx<0 and dy>0:
        imgout[:dx, :-dy, :] = img[-dx:, :-dy, :]
    if dx<0 and dy<0:
        imgout[:dx, :dy, :] = img[-dx:, -dy:, :]
    return imgout.astype('uint8')

# change contrast of image
def Change_contrast(img, level):
    factor = (259.0 * (level + 255)) / (255.0 * (259 - level))
    img_new = 128.0 + factor*(img-128.0)
    imgout = np.maximum(0, np.minimum(255, img_new))
    return imgout.astype('uint8')

def genRandNew(img):
    flag = np.random.randint(0,3,(1,))
    if flag==0:
        numbers = [-6,-5,-4,-3,-2,-1,1,2,3,4,5,6]
        dx = random.choice(numbers)
        dy = random.choice(numbers)
        imgnew = Shift(img, dx, dy)
    elif flag==1:
        degree = np.random.randint(-60, 60, (1,))
        imgnew = Rotation(img, degree)
    elif flag==2:
        level = np.random.randint(-80,80,(1,))
        imgnew = Change_contrast(img, level)
    return imgnew.reshape(1,img.shape[0],img.shape[1],img.shape[2])

def rgb2gray(rgb):
    gray = np.dot(rgb[...,:3], [0.299, 0.587, 0.114])

# setup tensorflow epochs and batch size
import tensorflow as tf
EPOCHS = 40
BATCH_SIZE = 100

# input size is 32x32x1 normalized image
from tensorflow.contrib.layers import flatten

# max pooling function
def maxpool2d(x, k=2):
    return tf.nn.max_pool(
        x,
        ksize=[1, k, k, 1],
        strides=[1, k, k, 1],
        padding='VALID')

# 2d convolution, x*W + b, with relu as activation
def conv2d(x, W, b, strides=1):
    x = tf.nn.conv2d(x, W, strides=[1, strides, strides, 1], padding='VALID')
    x = tf.nn.bias_add(x, b)
    return tf.nn.relu(x)

def CNNmodel(x):    
    # tf.truncated_normal parameters
    mu = 0
    sigma = 0.1
    
    # specify weights for each layer
    weights = {
        'wc1': tf.Variable(tf.truncated_normal(shape=[5,5,1,12], mean=mu, stddev=sigma)),
        'wc2': tf.Variable(tf.truncated_normal(shape=[5,5,12,32], mean=mu, stddev=sigma)),
        'wf3': tf.Variable(tf.truncated_normal(shape=[800, 200], mean=mu, stddev=sigma)),
        #'wf4': tf.Variable(tf.truncated_normal(shape=[120, 84], mean=mu, stddev=sigma)),
        #'wf5': tf.Variable(tf.truncated_normal(shape=[84, 43], mean=mu, stddev=sigma))
        'wf5': tf.Variable(tf.truncated_normal(shape=[200, 43], mean=mu, stddev=sigma))
    }
    
    biases = {
        'bc1': tf.Variable(tf.truncated_normal(shape=[12], mean=mu, stddev=sigma)),
        'bc2': tf.Variable(tf.truncated_normal(shape=[32], mean=mu, stddev=sigma)),
        'bf3': tf.Variable(tf.truncated_normal(shape=[200], mean=mu, stddev=sigma)),
        #'bf4': tf.Variable(tf.truncated_normal(shape=[84], mean=mu, stddev=sigma)),
        'bf5': tf.Variable(tf.truncated_normal(shape=[43], mean=mu, stddev=sigma))
    }
    
    
    # Layer 1: Convolutional. Input = 32x32x1. Output = 28x28x12. Activation relu
    conv1_out = conv2d(x, weights['wc1'], biases['bc1'])
    # Pooling. Input = 28x28x12. Output = 14x14x12.
    conv1 = maxpool2d(conv1_out, k=2)

    # Layer 2: Convolutional. Input = 14x14x12. Output = 10x10x32. Activation relu
    conv2_out = conv2d(conv1, weights['wc2'], biases['bc2'])
    # Pooling. Input = 10x10x32. Output = 5x5x32
    conv2 = maxpool2d(conv2_out, k=2)
    # Flatten. Input = 5x5x32. Output = 800.
    conv2_flat = flatten(conv2)
    
    # Layer 3: Dropout layer to avoid overfitting
    conv3 = tf.nn.dropout(conv2_flat, keep_prob=0.8) 

    # Layer 4: Fully Connected. Input = 800. Output = 200. Activation relu
    conv3 = tf.add(tf.matmul(conv3, weights['wf3']), biases['bf3'])
    conv4 = tf.nn.relu(conv3)

    # Layer 5: Fully Connected. Input = 200. Output = 43.
    logits = tf.add(tf.matmul(conv4, weights['wf5']), biases['bf5'])
    
    # note conv1, conv2, conv3, and conv4 are used to visualize the layer outputs
    return logits, conv1_out, conv2_out, conv3, conv4

## create placeholder for feature data input and one_hot encode label
x = tf.placeholder(tf.float32, (None, 32, 32, 1))
y = tf.placeholder(tf.int32, (None))
one_hot_y = tf.one_hot(y, 43)

## create gradient descent optimizer for training CNN
rate = 0.001
logits, conv1, conv2, conv3, conv4 = CNNmodel(x)
# calculate cross entropy and loss function
cross_entropy = tf.nn.softmax_cross_entropy_with_logits(labels=one_hot_y, logits=logits)
loss_operation = tf.reduce_mean(cross_entropy)
# setup gradient descent optimizer
optimizer = tf.train.AdamOptimizer(learning_rate = rate)
training_operation = optimizer.minimize(loss_operation)

## model validation, used to evaluate prediction accuracy
correct_prediction = tf.equal(tf.argmax(logits, 1), tf.argmax(one_hot_y, 1))
accuracy_operation = tf.reduce_mean(tf.cast(correct_prediction, tf.float32))
saver = tf.train.Saver()

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sign_count = np.zeros((43,))
    accurate_count = np.zeros((43,))
    sess = tf.get_default_session()
    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x,  y: batch_y})
        flag_accurate = correct_prediction.eval(session=sess,feed_dict={x: batch_x, y: batch_y})
        for i in range(len(batch_y)):
            sign_count[batch_y[i]] += 1
            if flag_accurate[i] == True:
                accurate_count[batch_y[i]] += 1
        total_accuracy += (accuracy * len(batch_x))
    accuracy_sign = accurate_count/sign_count
    #print(sign_count)
    #print(accurate_count)
    return total_accuracy / num_examples, accuracy_sign

Step 4 (Optional): Visualize the Neural Network's State with Test Images¶

This Section is not required to complete but acts as an additional excersise for understaning the output of a neural network's weights. While neural networks can be a great learning device they are often referred to as a black box. We can understand what the weights of a neural network look like better by plotting their feature maps. After successfully training your neural network you can see what it's feature maps look like by plotting the output of the network's weight layers in response to a test stimuli image. From these plotted feature maps, it's possible to see what characteristics of an image the network finds interesting. For a sign, maybe the inner network feature maps react with high activation to the sign's boundary outline or to the contrast in the sign's painted symbol.

Provided for you below is the function code that allows you to get the visualization output of any tensorflow weight layer you want. The inputs to the function should be a stimuli image, one used during training or a new one you provided, and then the tensorflow variable name that represents the layer's state during the training process, for instance if you wanted to see what the LeNet lab's feature maps looked like for it's second convolutional layer you could enter conv2 as the tf_activation variable.

For an example of what feature map outputs look like, check out NVIDIA's results in their paper End-to-End Deep Learning for Self-Driving Cars in the section Visualization of internal CNN State. NVIDIA was able to show that their network's inner weights had high activations to road boundary lines by comparing feature maps from an image with a clear path to one without. Try experimenting with a similar test to show that your trained network's weights are looking for interesting features, whether it's looking at differences in feature maps from images with or without a sign, or even what feature maps look like in a trained network vs a completely untrained one on the same sign image.
